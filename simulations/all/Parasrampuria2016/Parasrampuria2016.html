from typing import Dict

from sbmlsim.data import DataSet, load_pkdb_dataframe
from sbmlsim.fit import FitMapping, FitData
from sbmlutils.console import console

from pkdb_models.models.edoxaban.experiments.base_experiment import (
    EdoxabanSimulationExperiment,
)
from pkdb_models.models.edoxaban.experiments.metadata import Tissue, Route, Dosing, ApplicationForm, Health, Health, \
    Fasting, EdoxabanMappingMetaData, Coadministration

from sbmlsim.plot import Axis, Figure
from sbmlsim.simulation import Timecourse, TimecourseSim

from pkdb_models.models.edoxaban.helpers import run_experiments


class Parasrampuria2016(EdoxabanSimulationExperiment):
    """Simulation experiment of Parasrampuria2016."""


    #fraction_absorbed = {"Fasting": EdoxabanSimulationExperiment.fasting_map["fasted"]}
    colors = {
        "ED60, KET_edo_ket": "tab:blue",
        "ED60, ERY_edo_ery": "tab:orange",
        "ED60, CYC_edo_cyc": "tab:red",
        "ED60_edo_ery": "black",
        "ED60_edo_cyc": "black",
        "ED60_edo_ket": "black",
    }

    interventions = ["ED60, KET", "ED60, ERY", "ED60, CYC", "ED60"]

    groups = {
        "edo_ket": ["ED60, KET", "ED60"] ,
        "edo_ery": ["ED60, ERY", "ED60"],
        "edo_cyc": ["ED60, CYC", "ED60"]}

    dose =  60

    infos_pk = {
        "[Cve_edo]": "edoxaban",
        "[Cve_m4]": "M4",
       # "Aurine_edo": "edoxaban_urine",
    }



    def datasets(self) -> Dict[str, DataSet]:
        dsets = {}
        for fig_id in ["Fig2", "Fig3", "Fig4"]:
            df = load_pkdb_dataframe(f"{self.sid}_{fig_id}", data_path=self.data_path)
            for label, df_label in df.groupby("label"):
                dset = DataSet.from_df(df_label, self.ureg)
                if label.startswith("edoxaban"):
                        dset.unit_conversion("mean", 1 / self.Mr.edo)
                if label.startswith("M4_"):
                    dset.unit_conversion("mean", 1 / self.Mr.m4)
                dsets[label] = dset
        return dsets

    def simulations(self) -> Dict[str, TimecourseSim]:
        Q_ = self.Q_
        tcsims = {}
        #for group in self.groups:
        for intervention in self.interventions:
                dose = self.dose
                tcsims[f"{intervention}"] = TimecourseSim(
                    [Timecourse(
                        start=0,
                        end=97 * 60,  # [min]
                        steps=500,
                        changes={
                            **self.default_changes(),
                            "PODOSE_edo": Q_(dose, "mg"),
                        },
                    )]
                )
        return tcsims

    def fit_mappings(self) -> Dict[str, FitMapping]:
        mappings = {}
        for group in self.groups.keys():
            for intervention in self.groups[group]:
                coadministration = Coadministration.NONE
                if "KET" in intervention:
                    coadministration = Coadministration.KETOCONAZOLE
                elif "ERY" in intervention:
                    coadministration = Coadministration.ERYTHROMYCIN
                elif "CYC" in intervention:
                    coadministration = Coadministration.CYCLOSPORINE

                # PK
                for ks, sid in enumerate(self.infos_pk):
                    name = self.infos_pk[sid]
                    mappings[f"fm_{name}_{intervention}_{group}"] = FitMapping(
                        self,
                        reference=FitData(
                            self,
                            dataset=f"{name}_{intervention}_{group}",
                            xid="time",
                            yid="mean",
                            yid_sd="mean_sd",
                            count="count",
                        ),
                        observable=FitData(
                            self, task=f"task_{intervention}", xid="time", yid=sid,
                        ),
                        metadata=EdoxabanMappingMetaData(
                            tissue=Tissue.URINE if "urine" in name else Tissue.PLASMA,
                            route=Route.PO,
                            application_form=ApplicationForm.TABLET,
                            dosing=Dosing.SINGLE,
                            health=Health.HEALTHY,
                            fasting=Fasting.FASTED,
                            coadministration=coadministration
                        ),
                    )

        return mappings

    def figures(self) -> Dict[str, Figure]:
        return {
            **self.figure_pk(),
        }

    def figure_pk(self) -> Dict[str, Figure]:
        figures = {}
        for group in self.groups.keys():
            fig = Figure(
                experiment=self,
                sid=f"Fig1_{group}",
                num_cols=2,
                name=f"{self.__class__.__name__} (healthy)",
            )
            plots = fig.create_plots(
                xaxis=Axis(self.label_time, unit=self.unit_time), legend=True
            )
            plots[0].set_yaxis(self.label_edo_plasma, unit=self.unit_edo)
            plots[1].set_yaxis(self.label_m4_plasma, unit=self.unit_m4)

            for intervention in self.groups[group]:
                for ks, sid in enumerate(self.infos_pk):
                    name = self.infos_pk[sid]
                    # simulation
                    plots[ks].add_data(
                        task=f"task_{intervention}",
                        xid="time",
                        yid=sid,
                        label=intervention,
                        color=self.colors[f"{intervention}_{group}"],
                    )

                    # data
                    plots[ks].add_data(
                        dataset=f"{name}_{intervention}_{group}",
                        xid="time",
                        yid="mean",
                        yid_sd="mean_sd",
                        count="count",
                        label=intervention,
                        color=self.colors[f"{intervention}_{group}"],
                    )
            figures[fig.sid] = fig

        return figures


if __name__ == "__main__":
    run_experiments(Parasrampuria2016, output_dir=Parasrampuria2016.__name__)
